JavaScript Scope - Notes

Understanding Scope in JavaScript

In JavaScript, scope refers to the current context of execution, which determines the accessibility (visibility) of variables. There are two main types of scope:

  • Global Scope (var): Variables declared outside of any function or block can be accessed anywhere in the program.
  • Local / Block Scope (let, const): Variables declared inside a block {} or function can only be accessed within that block or function.

Example 1: Block vs Function Scope

if (true) {
  let blockScopedLet = "I am block scoped (let)";
  const blockScopedConst = "I am block scoped (const)";
  var globalScopedVar = "I am function scoped (var)";

  console.log(blockScopedLet);   // Accessible
  console.log(blockScopedConst); // Accessible
  console.log(globalScopedVar);  // Accessible
}

console.log(blockScopedLet);   // ❌ Not Accessible
console.log(blockScopedConst); // ❌ Not Accessible
console.log(globalScopedVar);  // ✅ Accessible
Note: let and const are block-scoped, while var is function-scoped.

Example 2: Global and Local Scope

var globalVar = "I am a global variable";

function testScope() {
  var localVar = "I am a local variable";
  console.log(globalVar); // Accessible
  console.log(localVar);  // Accessible
}

testScope();

console.log(globalVar); // Accessible
console.log(localVar);  // ❌ Not Accessible

Example 3: var, let, const Behavior

if (true) {
  const a = 10;
  let b = 20;
  var c = 30;

  console.log(a); // 10
  console.log(b); // 20
  console.log(c); // 30
}

console.log(a); // ❌ Not Accessible
console.log(b); // ❌ Not Accessible
console.log(c); // ✅ Accessible
Tip: Always prefer let or const over var to avoid unexpected access or hoisting issues.
