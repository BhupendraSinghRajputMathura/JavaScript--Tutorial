Object in JavaScript

  1. Defination:

    Collection of key value pair and each key is distinct

  2. Declaration:

                        const obj1 = {id:1, age:21, name:"aadi"};

    //OR

                        const obj1 = {
                        id:1,
                        age:21,
                        name:"aadi"
                    };

  3. Accessing:

                        const obj1 = {
                        id:1,
                        age:21,
                        name:"aadi"
                    };

                    console.log(obj1.id); // 1

                    
    //OR
    

                    console.log(obj1["id"]); // 1

  4. Over Writing

                        const obj1 = {
                        id:1,
                        age:21,
                        name:"aadi"
                    };
                    
                    obj1.id = 2;
                    obj1.age = 22;
                    obj1.name ="Singh";

                    console.log(obj1.id); // 2
                    console.log(obj1.age); // 22
                    console.log(obj1.name); // Singh

  5. Object.freeze(object_name)

    Used to lock the given object then it become immutable(Unchangable)

                        const obj1 = {
                        id:1,
                        age:21,
                        name:"aadi"
                    };

                    Object.freeze(obj1);
                    
                    obj1.id = 2;
                    obj1.age = 22;
                    obj1.name ="Singh";

                    console.log(obj1.id); // 1
                    console.log(obj1.age); // 21
                    console.log(obj1.name); // aadi

  6. Object.keys(object_name)

    Used to get keys of object

                        const obj1 = {
                        id:1,
                        age:21,
                        name:"aadi"
                    };

                    const key=Object.keys(obj1);
                    console.log(key); // ['id','age','name']

  7. Object.value(object_name)

    Used to get values of object

                        const obj1 = {
                        id:1,
                        age:21,
                        name:"aadi"
                    };

                    const values=Object.value(obj1);
                    console.log(values); // ['1','21','aadi']

  8. Object.entries(object_name)

    Used to get entries from object

                        const obj1 = {
                        id:1,
                        age:21,
                        name:"aadi"
                    };

                    const entities=Object.entries(obj1);
                    console.log(entities); // [ [ 'id', 1 ], [ 'age', 21 ], [ 'name', 'aadi' ] ]

  9. Spread Operator(...)

    Used to spread elements

    • Copy Object

                                  const obj1 = {a:1, b:2};
                            
                            const copy = {...obj1};
                            console.log(copy); // { a: 1, b: 2}

    • Merging

                                  const obj1 = {a:1, b:2};
                            const obj2 = {c:3, d:4};
                            const obj3 = {e:5, f:6};
                            
                            const merged = {...obj1,...obj2,...obj3};
                            console.log(merged); // { a: 1, b: 2, c: 3, d: 4, e: 5, f: 6 }

    • Override

                                  const obj1 = {a:1, b:2};
                            
                            const update = {...obj1,b:5};
                            console.log(merged); // { a: 1, b: 5 }

    • Nested Object (Shallow copy issue)

                                  const student = {
                                name: "Raj",
                                address: { city: "Mathura", pincode: 281001 }
                            };

                            const newStudent = { ...student };
                            newStudent.address.city = "Delhi";
                            
                            console.log(student.address.city); // Delhi

  10. De-Structuring of Object

                        const obj = {
                        id:1,
                        age:21,
                        name:"Aadi"
                    };

                    console.log(obj.id);  // 1
                    console.log(obj.age);  //21

                // De-Structuring

                    const {id, age , name} = obj;

                    console.log(id);  // 1
                    console.log(age);  //21
                    console.log(name);  // Aadi
