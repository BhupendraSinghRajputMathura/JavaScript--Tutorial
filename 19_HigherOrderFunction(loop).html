Higher Order Functions in JavaScript

Higher order functions are functions that can take other functions as arguments or return them as output. They are a key feature of functional programming and allow for more abstract and flexible code.

Examples

1. Functions as First-Class Citizens

function greet(name) {
    return "Hello, " + name;
}

function processUserInput(callback) {
    var name = prompt("Please enter your name.");
    alert(callback(name));
}

processUserInput(greet);

2. Array Methods

const numbers = [1, 2, 3, 4, 5];

const doubled = numbers.map(function(num) {
    return num * 2;
});

const filtered = numbers.filter(function(num) {
    return num > 2;
});

3. Returning Functions

function makeCounter() {
    let count = 0;
    return function() {
        count++;
        return count;
    };
}

const counter = makeCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

Common Higher Order Functions

  • forEach(): Executes a provided function once for each array element.
  • filter(): Filters elements in an array based on a condition and returns a new array.
  • map(): Transforms each element in an array and returns a new array.
  • reduce(): Reduces an array to a single value by applying a function to each element.

Benefits of Higher Order Functions

  • Encourages code reusability and modularity.
  • Makes code more readable and expressive.
  • Facilitates functional programming techniques.

.forEach()

It is a built-in higher order function method and it does not return anything(Always undefined)

The forEach() method executes a provided function once for each array element. It is commonly used to iterate over arrays.

syntax


        array.forEach(function(currentValue, index, arr){
            //your code here
        })

        
OR


        array.forEach((currentValue, index, arr) => {
            //your code here
        })

Example

const arr = [1, 2, 3, 4, 5];
arr.forEach((element) => {
    console.log(element);
});
// Output:
// 1
// 2
// 3
// 4
// 5

    
const fruits = ['apple', 'banana', 'cherry'];
fruits.forEach((fruit, index) => {
    console.log(index + ': ' + fruit);
});
// Output:
// 0: apple
// 1: banana
// 2: cherry
    


    const objectArray = [
        { name: 'Alice', age: 25 },
        { name: 'Bob', age: 30 },
        { name: 'Charlie', age: 35 }
    ];
    objectArray.forEach((obj) => {
        console.log(obj.name + ' is ' + obj.age + ' years old.');
    });
    // Output:
    // Alice is 25 years old.
    // Bob is 30 years old.
    // Charlie is 35 years old.

.filter()

It is a built-in higher order function method and it always return a new array

The filter() method creates a new array and It is commonly used to filter elements from an array based on specific criteria.

syntax


    array.filter(function(currentValue, index, arr){
        //your code here
    })

    
OR


    array.filter((currentValue, index, arr) => {
        //your code here
    })

Example

const numbers = [1, 2, 3, 4, 5];
const evenNumbers = numbers.filter((num) => num % 2 === 0);
console.log(evenNumbers); // Output: [2, 4]


    const mixedNumbers = [10, 15, 20, 25, 30];
    const greaterThan20 = mixedNumbers.filter((num) => num > 20);
    console.log(greaterThan20); // Output: [25, 30]


    const words = ['apple', 'banana', 'cherry', 'date'];
    const longWords = words.filter((word) => word.length > 5);
    console.log(longWords); // Output: ['banana', 'cherry']


    const objectArray = [
        { name: 'Alice', age: 25 },
        { name: 'Bob', age: 30 },
        { name: 'Charlie', age: 35 }
    ];
    const adults = objectArray.filter((obj) => obj.age >= 30);
    console.log(adults);
    // Output:
    // [ { name: 'Bob', age: 30 }, { name: 'Charlie', age: 35 } ]


    const students = [
        { name: 'Alice', score: 85 },
        { name: 'Bob', score: 92 },
        { name: 'Charlie', score: 78 }
    ];
    const passedStudents = students.filter((student) => student.score >= 80);
    console.log(passedStudents);
    // Output:
    // [ { name: 'Alice', score: 85 }, { name: 'Bob', score: 92 } ]

    
    const products = [
        { name: 'Laptop', price: 1000 },
        { name: 'Phone', price: 500 },
        { name: 'Tablet', price: 300 }
    ];
    const expensiveProducts = products.filter((product) => product.price > 400);
    console.log(expensiveProducts);
    // Output:
    // [ { name: 'Laptop', price: 1000 }, { name: 'Phone', price: 500 } ]
    


    const cities = ['New York', 'Los Angeles', 'Chicago', 'Houston', 'Phoenix'];
    const citiesWithO = cities.filter((city) => city.includes('o'));
    console.log(citiesWithO); // Output: ['Los Angeles', 'Houston']


    const name = ['Alice', 'Bob', 'Charlie', 'David'];
    const namesWithA = name.filter((n) => n.startsWith('A'));
    console.log(namesWithA); // Output: ['Alice']
    const nameWitha = name.filter((n) => n.toLowerCase().includes('a'));
    console.log(nameWitha); // Output: ['Alice', 'Charlie', 'David']

.map()

It is a built-in higher order function method and it always return a new array

It is commonly used to transform elements in an array.

The map() method creates a new array populated with the results of calling a provided function on every element in the calling array.

syntax


    array.map(function(currentValue, index, arr){
        //your code here
    })

    
OR


    array.map((currentValue, index, arr) => {
        //your code here
    })

Example

const numbers = [1, 2, 3, 4, 5];
const squared = numbers.map((num) => num * num);
console.log(squared); // Output: [1, 4, 9, 16, 25]


    const mixedNumbers = [1, 2, 3, 4, 5];
    const incremented = mixedNumbers.map((num) => num + 1);
    console.log(incremented); // Output: [2, 3, 4, 5, 6]


    const words = ['hello', 'world', 'javascript'];
    const uppercased = words.map((word) => word.toUpperCase());
    console.log(uppercased); // Output: ['HELLO', 'WORLD', 'JAVASCRIPT']


    const objectArray = [
        { name: 'Alice', age: 25 },
        { name: 'Bob', age: 30 },
        { name: 'Charlie', age: 35 }
    ];
    const names = objectArray.map((obj) => obj.name);
    console.log(names); // Output: ['Alice', 'Bob', 'Charlie']


    const students = [
        { name: 'Alice', score: 85 },
        { name: 'Bob', score: 92 },
        { name: 'Charlie', score: 78 }
    ];
    const studentNames = students.map((student) => student.name);
    console.log(studentNames); // Output: ['Alice', 'Bob', 'Charlie']


    const products = [
        { name: 'Laptop', price: 1000 },
        { name: 'Phone', price: 500 },
        { name: 'Tablet', price: 300 }
    ];
    const productNames = products.map((product) => product.name);
    console.log(productNames); // Output: ['Laptop', 'Phone', 'Tablet']


    const cities = ['New York', 'Los Angeles', 'Chicago', 'Houston', 'Phoenix'];
    const cityLengths = cities.map((city) => city.length);
    console.log(cityLengths); // Output: [8, 11, 7, 7, 7]
