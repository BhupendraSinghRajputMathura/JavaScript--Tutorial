Operators in JavaScript

1. Arithmetic Operators

console.log(5 + 3); // 8 console.log(5 - 3); // 2 console.log(5 * 3); // 15 console.log(5 / 2); // 2.5 console.log(5 % 2); // 1 console.log(5 ** 2); // 25 console.log(5 + "3"); // "53" console.log("5" - 3); // 2 console.log("5" * 3); // 15

2. Assignment Operators

let a = 5; a += 3; // 8 a -= 2; // 6 a *= 2; // 12 a /= 3; // 4 a %= 3; // 1 a **= 3; // 1 console.log(a);

3. Increment & Decrement

let b = 5; console.log(b++); // 5 console.log(++b); // 7 console.log(b--); // 7 console.log(--b); // 5

4. Logical Operators

console.log(true && false); // false console.log(true || false); // true console.log(!true); // false

5. Ternary Operator

let age = 20; let canVote = (age >= 18) ? "Yes" : "No"; console.log(canVote); // "Yes"

6. typeof Operator

console.log(typeof 5); // number console.log(typeof "hello"); // string console.log(typeof null); // object console.log(typeof []); // object console.log(typeof function(){}); // function

7. Bitwise Operators

console.log(5 & 3); // 1 console.log(5 | 3); // 7 console.log(5 ^ 3); // 6 console.log(~5); // -6 console.log(5 << 1); // 10 console.log(5 >> 1); // 2

8. Equality & Relational Operators

console.log(5 == "5"); // true console.log(5 === "5"); // false console.log(null == undefined); // true console.log(null === undefined); // false console.log(5 > 3); // true console.log("abc" > "ab"); // true

9. Nullish Coalescing Operator (??)

let foo = null; let bar = foo ?? "default"; console.log(bar); // "default"

10. Optional Chaining Operator (?.)

let obj = { a: { b: { c: 42 } } }; console.log(obj?.a?.b?.c); // 42 console.log(obj?.x?.y?.z); // undefined

11. Spread Operator (...)

let arr1 = [1, 2, 3]; let arr2 = [...arr1, 4, 5]; console.log(arr2); // [1, 2, 3, 4, 5]

12. Destructuring Assignment

let [x, y] = [1, 2]; console.log(x, y); let { a, b } = { a: 3, b: 4 }; console.log(a, b);
